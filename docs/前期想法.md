### 二、 核心动作识别原理（以 MediaPipe 为例）

MediaPipe 会返回人体 33 个关键点的 `(x, y, z)` 坐标。你可以通过计算这些点之间的**向量夹角**来判断动作是否标准。

**例如：判断“深蹲” (Squat)**
1. 提取关键点：髋部 (Hip, ID 23/24)、膝部 (Knee, ID 25/26)、踝部 (Ankle, ID 27/28)。
2. 计算向量：向量 A = 膝部 - 髋部，向量 B = 膝部 - 踝部。
3. 计算夹角：使用点积公式计算向量 A 和 B 的夹角。
4. 判定逻辑：站立时夹角约 160°-180°，深蹲到底时夹角小于 90°。当检测到角度从 >150° 变为 <90° 再回到 >150°，即计为一次有效深蹲。

### 三、 关卡设计灵感

1. **关卡 1：热身 - 手臂上举 (Arm Raises)**
   - **目标**：10 次
   - **检测**：手腕 (Wrist) 的 Y 坐标高于肩膀 (Shoulder) 的 Y 坐标。
2. **关卡 2：下肢 - 标准深蹲 (Squats)**
   - **目标**：15 次
   - **检测**：膝关节角度 < 90°，且躯干保持相对直立（肩部和髋部的 X 坐标偏移量在合理范围内）。
3. **关卡 3：心肺 - 开合跳 (Jumping Jacks)**
   - **目标**：20 次
   - **检测**：双手腕距离大于肩宽的 1.5 倍，且双脚踝距离大于骨盆宽度。
4. **关卡 4：核心 - 躯干扭转 (Torso Twists)**
   - **目标**：30 秒内尽可能多次
   - **检测**：左肩和右肩的 Z 轴（深度）差值发生周期性正负交替，说明身体在左右转动。

### 四、 快速原型代码示例 (Python + MediaPipe)

这是一个最小可行性产品 (MVP) 的代码，它会打开摄像头，实时检测你的**深蹲**动作，并在屏幕上计数。你可以直接运行它来验证想法。

**前置准备**:


**原型代码**:
```python
import cv2
import mediapipe as mp
import numpy as np
import math

# 初始化 MediaPipe Pose
mp_pose = mp.solutions.pose
pose = mp_pose.Pose(min_detection_confidence=0.5, min_tracking_confidence=0.5)
mp_drawing = mp.solutions.drawing_utils

# 计算两个向量之间的夹角
def calculate_angle(a, b, c):
    a = np.array(a) # 第一个点
    b = np.array(b) # 顶点 (例如膝盖)
    c = np.array(c) # 第三个点
    
    radians = np.arctan2(c[1]-b[1], c[0]-b[0]) - np.arctan2(a[1]-b[1], a[0]-b[0])
    angle = np.abs(radians * 180.0 / np.pi)
    if angle > 180.0:
        angle = 360 - angle
    return angle

# 启动摄像头
cap = cv2.VideoCapture(0)
squat_count = 0
stage = "up" # 状态机：up (站立) 或 down (蹲下)

print("按 'q' 键退出")

while cap.isOpened():
    ret, frame = cap.read()
    if not ret:
        break

    # 转换颜色空间 (BGR to RGB)
    image = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
    image.flags.writeable = False
    results = pose.process(image)
    
    # 转换回 BGR 以便 OpenCV 显示
    image.flags.writeable = True
    image = cv2.cvtColor(image, cv2.COLOR_RGB2BGR)
    
    # 提取关键点
    try:
        landmarks = results.pose_landmarks.landmark
        
        # 获取右侧身体关键点坐标 [x, y]
        hip = [landmarks[mp_pose.PoseLandmark.RIGHT_HIP.value].x, landmarks[mp_pose.PoseLandmark.RIGHT_HIP.value].y]
        knee = [landmarks[mp_pose.PoseLandmark.RIGHT_KNEE.value].x, landmarks[mp_pose.PoseLandmark.RIGHT_KNEE.value].y]
        ankle = [landmarks[mp_pose.PoseLandmark.RIGHT_ANKLE.value].x, landmarks[mp_pose.PoseLandmark.RIGHT_ANKLE.value].y]
        
        # 计算膝关节角度
        angle = calculate_angle(hip, knee, ankle)
        
        # 动作判定逻辑 (深蹲)
        if angle < 100: # 蹲下状态
            stage = "down"
        if angle > 140 and stage == "down": # 站起状态，完成一次
            stage = "up"
            squat_count += 1
            print(f"深蹲次数: {squat_count}")
            
        # 在屏幕上绘制角度和计数
        cv2.putText(image, f'Angle: {int(angle)}', 
                    tuple(np.multiply(knee, [640, 480]).astype(int)), 
                    cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 2, cv2.LINE_AA)
        cv2.putText(image, f'Squats: {squat_count}', 
                    (50, 50), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 3, cv2.LINE_AA)
        cv2.putText(image, f'Stage: {stage}', 
                    (50, 100), cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 0, 0), 3, cv2.LINE_AA)
                    
    except Exception as e:
        pass

    # 绘制骨骼关键点
    if results.pose_landmarks:
        mp_drawing.draw_landmarks(image, results.pose_landmarks, mp_pose.POSE_CONNECTIONS,
                                  mp_drawing.DrawingSpec(color=(245,117,66), thickness=2, circle_radius=2),
                                  mp_drawing.DrawingSpec(color=(245,66,230), thickness=2, circle_radius=2))

    # 显示画面
    cv2.imshow('Exercise Game Prototype', image)

    if cv2.waitKey(10) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
```

#### 2. 核心 Hook：动作捕捉与游戏逻辑 (`usePoseGame.ts`)
这是游戏的大脑。我们使用 MediaPipe 最新的 `PoseLandmarker`，并结合状态机来判断深蹲。

```typescript
import { useEffect, useRef, useState, useCallback } from 'react';
import { PoseLandmarker, FilesetResolver, DrawingUtils } from '@mediapipe/tasks-vision';

export const usePoseGame = (videoRef: React.RefObject<HTMLVideoElement>, canvasRef: React.RefObject<HTMLCanvasElement>) => {
  const [squatCount, setSquatCount] = useState(0);
  const [stage, setStage] = useState<'up' | 'down'>('up');
  const poseLandmarkerRef = useRef<PoseLandmarker | null>(null);
  const rafIdRef = useRef<number | null>(null);

  // 初始化 MediaPipe
  useEffect(() => {
    const createPoseLandmarker = async () => {
      const vision = await FilesetResolver.forVisionTasks(
        "https://cdn.jsdelivr.net/npm/@mediapipe/tasks-vision@latest/wasm"
      );
      poseLandmarkerRef.current = await PoseLandmarker.createFromOptions(vision, {
        baseOptions: {
          modelAssetPath: `https://storage.googleapis.com/mediapipe-models/pose_landmarker/pose_landmarker_lite/float16/1/pose_landmarker_lite.task`,
          delegate: "GPU" // 优先使用 GPU 加速
        },
        runningMode: "VIDEO",
        numPoses: 1
      });
      console.log("MediaPipe 模型加载完成");
    };
    createPoseLandmarker();
  }, []);

  // 计算向量夹角 (用于判断深蹲)
  const calculateAngle = (a: any, b: any, c: any) => {
    const radians = Math.atan2(c.y - b.y, c.x - b.x) - Math.atan2(a.y - b.y, a.x - b.x);
    let angle = Math.abs(radians * 180.0 / Math.PI);
    if (angle > 180.0) angle = 360 - angle;
    return angle;
  };

  // 核心渲染与推理循环
  const predict = useCallback(() => {
    if (!poseLandmarkerRef.current || !videoRef.current || !canvasRef.current) {
      rafIdRef.current = requestAnimationFrame(predict);
      return;
    }

    const video = videoRef.current;
    const canvas = canvasRef.current;
    const ctx = canvas.getContext('2d')!;

    // 设置 Canvas 尺寸与视频一致
    if (canvas.width !== video.videoWidth) {
      canvas.width = video.videoWidth;
      canvas.height = video.videoHeight;
    }

    const startTimeMs = performance.now();
    const landmarks = poseLandmarkerRef.current.detectForVideo(video, startTimeMs);

    // 清空画布并绘制视频
    ctx.save();
    // ⚠️ 跨平台关键点：前置摄像头需要水平翻转，符合照镜子直觉
    ctx.scale(-1, 1); 
    ctx.translate(-canvas.width, 0);
    ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
    ctx.restore();

    if (landmarks.landmarks && landmarks.landmarks.length > 0) {
      const pose = landmarks.landmarks[0];
      
      // 绘制骨骼 (MediaPipe 官方工具)
      const drawingUtils = new DrawingUtils(ctx);
      ctx.save();
      ctx.scale(-1, 1);
      ctx.translate(-canvas.width, 0);
      drawingUtils.drawLandmarks(pose, { color: '#FF0000', radius: 2 });
      drawingUtils.drawConnectors(pose, PoseLandmarker.POSE_CONNECTIONS, { color: '#00FF00', lineWidth: 3 });
      ctx.restore();

      // 提取右侧髋、膝、踝 (注意：因为镜像翻转，实际取左侧点或右侧点需根据需求调整，这里以右侧为例)
      // MediaPipe 返回的是归一化坐标 (0-1)
      const hip = pose[24]; // Right Hip
      const knee = pose[26]; // Right Knee
      const ankle = pose[28]; // Right Ankle

      const angle = calculateAngle(hip, knee, ankle);

      // 游戏逻辑：深蹲状态机
      if (angle < 100 && stage === 'up') {
        setStage('down');
      } else if (angle > 140 && stage === 'down') {
        setStage('up');
        setSquatCount(prev => prev + 1);
      }
    }

    rafIdRef.current = requestAnimationFrame(predict);
  }, [stage]);

  // 启动/停止循环
  useEffect(() => {
    rafIdRef.current = requestAnimationFrame(predict);
    return () => {
      if (rafIdRef.current) cancelAnimationFrame(rafIdRef.current);
    };
  }, [predict]);

  return { squatCount, stage };
};
```

```tsx
import { useEffect, useRef, useState } from 'react';
import { usePoseGame } from './usePoseGame';

function App() {

      {/* 隐藏的 Video 元素 */}

      {/* 游戏画布 */}
      <canvas 
        ref={canvasRef} 
        className="max-w-full max-h-full object-contain" 
      />

      {/* UI 覆盖层 */}
      {isCameraReady && (

        <div className="absolute inset-0 flex items-center justify-center text-white text-xl">
          正在初始化摄像头...
        </div>
      )}
    </div>
  );
}

export default App;
```

### 三、 跨平台 (Web) 核心避坑指南 (极其重要)

做 Web 体感游戏，最容易在以下四个地方翻车，请务必注意：

#### 1. HTTPS 是强制的
浏览器的安全策略规定，**除了 `localhost`，任何调用摄像头 (`getUserMedia`) 的网页必须在 HTTPS 环境下运行**。
*   **开发时**：使用 `localhost` 没问题。
*   **部署时**：必须配置 SSL 证书。推荐使用 Vercel / Netlify / GitHub Pages，它们免费提供 HTTPS。

#### 2. 镜像与坐标系问题 (90% 的 Bug 来源)
*   **现象**：用户举左手，屏幕上的骨骼举右手。
*   **原因**：前置摄像头拍到的画面默认是**镜像**的（像照镜子），但 MediaPipe 输出的坐标是**真实世界坐标**。
*   **解决**：在 Canvas 绘制时，必须对画面和骨骼进行水平翻转（如上面代码中的 `ctx.scale(-1, 1)`）。**注意：计算角度时，使用的是原始未翻转的坐标，绘制时才翻转。**

#### 3. 移动端性能与发热
手机浏览器跑 MediaPipe 会比 PC 吃力，容易导致手机发热降频、掉帧。
*   **优化 1**：限制视频分辨率。在 `getUserMedia` 中设置 `width: 640, height: 480` 即可，不需要 1080P，640P 足够识别骨骼。
*   **优化 2**：跳帧推理。不需要每一帧都跑 MediaPipe。可以设置一个计数器，每 2 帧或 3 帧推理一次，中间帧直接绘制画面。
*   **优化 3**：使用 `delegate: "GPU"` (如代码所示)，利用手机的 WebGPU/WebGL 加速。

#### 4. iOS Safari 的特殊限制
*   **自动播放**：iOS 不允许视频自动播放带声音的内容。必须给 `<video>` 标签加上 `muted` 和 `playsInline` 属性（代码中已包含）。
*   **音频播放**：如果你想做“做对了一个动作，播放‘叮’的音效”，在 iOS 上第一次播放音效必须由**用户的触摸事件**（如点击“开始游戏”按钮）触发，不能在游戏过程中突然自动播放。
